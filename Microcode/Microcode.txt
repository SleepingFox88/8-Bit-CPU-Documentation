;Header instruction, executed at the start of all micro programs
Header:
    fromRAM, toID, incIP

HLT:
    HLT

JMP_const:
    if(ujump), incIP, fromRAM, endMP

MOV_reg_const:
    fromRAM, toRegWrite, incIP
    fromRAM, toWriteReg, incIP, endMP
    
MOV_reg_reg:
    fromRAM, toRegWrite, incIP
    fromRAM, toRegRead
    fromReg, toWriteReg, incIP, endMP
